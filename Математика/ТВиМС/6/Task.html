def calculate_psi(observed1, observed2, bins=30):
    bin_edges = np.histogram(np.concatenate([observed1, observed2]), bins=bins)[1]

    hist1 = np.histogram(observed1, bins=bin_edges)[0]
    hist2 = np.histogram(observed2, bins=bin_edges)[0]

    rel_freq1 = hist1 / len(observed1)
    rel_freq2 = hist2 / len(observed2)

    psi_values = np.sum(rel_entr(rel_freq1, rel_freq2) + rel_entr(rel_freq2, rel_freq1))

    return psi_values

Теперь для каждой недели кроме первой будем сравнивать её с предыдущей и с первой, используя PSI.
